1. Table name = users
    
    CREATE TABLE users
    (
        userId VARCHAR(50),
        username VARCHAR(10),
        password VARCHAR(30)
    ) ENGINE = InnoDB;

2. Table name = units

    CREATE TABLE units
    (
        unitId VARCHAR(50),
        addedById VARCHAR(50),
        unit_symbol VARCHAR(50),
        unit_full_form VARCHAR(50)
    ) ENGINE = InnoDB;
